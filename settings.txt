Install via Homebrew:
brew install pinentry-mac

Then configure GPG:
echo "pinentry-program /usr/local/bin/pinentry-mac" >> ~/.gnupg/gpg-agent.conf
killall gpg-agent

defaults write com.apple.finder _FXShowPosixPathInTitle -bool YES

mousewheel.default.delta_multiplier_y 190
mousewheel.with_meta.action 3

do shell script "/Applications/Firefox.app/Contents/MacOS/firefox -p main &"

emoji picker animation disable:
defaults write -g NSAutomaticWindowAnimationsEnabled -bool false

cd /Applications/kolourpaint.app/Contents/MacOS
mv kolourpaint kolourpaint.bin
nano kolourpaint

export QT_STYLE_OVERRIDE=Fusion
